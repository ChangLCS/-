CREATE TABLE `mdm_enterprise_rel` (
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'id编码',
  `member_id` int(11) DEFAULT '0' COMMENT '会员id，哪个会员提交的审核',
  `enterprise_id` int(11) DEFAULT '0' COMMENT '机构表id',
  `audit_id` int(11) NOT NULL COMMENT '审核表id,根据处理对象不同指向不同的审核表',
  `audit_type` int(11) NOT NULL COMMENT '申请类型 1:新增 2：修改 3：删除',
  `process_object` int(11) NOT NULL COMMENT '申请对象 0:基本信息 1：工信部排名 2:销售额 3：配送区域 4：仓储设施 5:运输车辆 6：企业诚信',
  `audit_status` int(11) NOT NULL COMMENT '审核状态',
  `remarks` varchar(256) DEFAULT NULL COMMENT '备注',
  `sort_code` int(11) DEFAULT NULL COMMENT '排序码',
  `create_by` varchar(20) DEFAULT NULL COMMENT '创建人',
  `create_time` datetime NOT NULL COMMENT '创建时间',
  `modify_by` varchar(20) DEFAULT NULL COMMENT '修改人',
  `modify_time` datetime DEFAULT NULL COMMENT '修改时间',
  `audit_by` varchar(20) DEFAULT NULL COMMENT '审核人',
  `audit_time` datetime DEFAULT NULL COMMENT '审核时间',
  `version` int(11) NOT NULL DEFAULT '0' COMMENT '版本号',
  `enabled` int(11) NOT NULL DEFAULT '1' COMMENT '启用状态(1启用0禁用)',
  `enterprise_type` int(11) NOT NULL DEFAULT '1',
  `enterprise_name` varchar(255) DEFAULT NULL,
  `enterprise_code` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `enterprise_id` (`enterprise_id`)
) ENGINE=InnoDB AUTO_INCREMENT=497 DEFAULT CHARSET=utf8 COMMENT='机构与关联关系审核表';
